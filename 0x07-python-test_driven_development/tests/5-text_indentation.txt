"""
test  cases for text identation function
"""

def test_text_indentation():
    from text_indentation import text_indentation

    # Test for valid input
    assert text_indentation("Hello") == "Hello"

    # Test for spacing after character .
    assert text_indentation("Rose.loki") == "Rose\n\nloki"

    # Test for character ?
    assert text_indentation("Rose?loki") == "Rose\n\nloki"

    # Test for character :
    assert text_indentation("rose:loki") == "rose\n\nloki"

    # Test for int
    try:
        text_indentation(5)
    except TypeError as e:
        assert str(e) == "text must be a string"

    # Test for not string
    try:
        text_indentation(None)
    except TypeError as e:
        assert str(e) == "text must be a string"

if __name__ == "__main__":
    test_text_indentation()
